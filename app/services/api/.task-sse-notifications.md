SSE stream handler:
```go
func (h *Handler) Stream(w http.ResponseWriter, r *http.Request) {
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")
    w.Header().Set("X-Accel-Buffering", "no") // nginx/cloudflare
    
    flusher, ok := w.(http.Flusher)
    if !ok { http.Error(...); return }
    
    userID := ctxutil.GetUserUUID(r.Context())
    ch := h.broker.Subscribe(userID)
    defer h.broker.Unsubscribe(userID, ch)
    
    // Send initial unread count
    count, _ := h.service.CountUnread(r.Context(), userID)
    sendSSE(w, flusher, Event{Type: "badge_update", Data: count})
    
    for {
        select {
        case event := <-ch:
            sendSSE(w, flusher, event)
        case <-r.Context().Done():
            return
        case <-time.After(30 * time.Second):
            // Heartbeat to keep connection alive
            fmt.Fprintf(w, ": heartbeat\n\n")
            flusher.Flush()
        }
    }
}
```

CREATE TABLE IF NOT EXISTS treasury.notifications (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id     UUID NOT NULL,
    title       VARCHAR(200) NOT NULL,
    message     TEXT NOT NULL,
    category    VARCHAR(50) NOT NULL,
